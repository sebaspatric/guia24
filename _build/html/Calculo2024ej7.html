import numpy as np

# Sucesi贸n Fibonacci

    x_an = n_vals[:N_max]
    y_an = fib[:N_max]

    x_an1 = n_vals[1:N_max] if N_max>1 else []
    y_an1 = fib[:N_max-1] if N_max>1 else []

    x_an2 = n_vals[2:N_max] if N_max>2 else []
    y_an2 = fib[:N_max-2] if N_max>2 else []

    step = dict(
        method='update',
        args=[{'x':[x_an, x_an1, x_an2],
               'y':[y_an, y_an1, y_an2]}],
        label=str(N_max)
    )
    steps.append(step)

sliders = [dict(active=18, pad={"t":50}, steps=steps, currentvalue={"prefix":"N_max = "})]

# Layout
fig.update_layout(
    autosize=True,
    width=None,
    height=None,
    sliders=sliders,
    title='Sucesi贸n Fibonacci con a_{n-1} y a_{n-2}',
    xaxis_title='n',
    yaxis_title='Valor',
    template='plotly_white'
)
fig.write_html(
    r'_build/html/7sucesion_fibonacci.html',
    include_plotlyjs='cdn',
    full_html=False
