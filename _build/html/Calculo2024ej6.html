Calculo2024#

Pregunta6#

6.

Si \(f(x)\) es una función continua en \(\mathbb{R}\) y se sabe que:

\[ \int_1^7 f(x) \, dx = 10, \quad \text{y} \quad \int_4^7 f(x) \, dx = 4, \]

entonces el valor de

\[ \int_1^4 3f(x) \, dx \]

es:

Desarrollo#

y se busca:

\[ \int_1^4 3f(x) \, dx = ? \]

Paso 1: Usar la propiedad de la integral definida sobre intervalos consecutivos#

Sabemos que:

\[ \int_1^7 f(x) \, dx = \int_1^4 f(x) \, dx + \int_4^7 f(x) \, dx. \]

Sustituyendo los valores:

\[ 10 = \int_1^4 f(x) \, dx + 4. \]

Por lo tanto,

\[ \int_1^4 f(x) \, dx = 10 - 4 = 6. \]

Paso 2: Calcular \(\int_1^4 3f(x) \, dx\)#

Multiplicando la integral por 3:

\[ \int_1^4 3f(x) \, dx = 3 \int_1^4 f(x) \, dx = 3 \times 6 = 18. \]

Respuesta:#

\[ \boxed{18} \]

La alternativa correcta es la c) 18.

import numpy as np
import plotly.graph_objects as go
from scipy.integrate import quad

# Función
def f(x):
    return np.sin(x) + 1.5

# Rango global
x = np.linspace(0, 8, 400)

# Función que genera los trazos de la figura para un conjunto de parámetros
def make_traces(x1, x2, x3, factor):
    x_fill_1_2 = np.linspace(x1, x2, 200)
    x_fill_2_3 = np.linspace(x2, x3, 200)
    x_fill_1_3 = np.linspace(x1, x3, 200)

    # Integrales
    area_1_2, _ = quad(f, x1, x2)
    area_2_3, _ = quad(f, x2, x3)
    area_1_3, _ = quad(f, x1, x3)
    area_1_2_factor = factor * area_1_2

    return [
        go.Scatter(x=x, y=f(x), mode="lines", name="f(x)", line=dict(color="blue")),
        go.Scatter(x=np.concatenate([[x1], x_fill_1_2, [x2]]),
                   y=np.concatenate([[0], f(x_fill_1_2), [0]]),
                   fill="toself", fillcolor="orange", opacity=0.5,
                   line=dict(color="orange"),
                   name=f"∫{x1:.0f}^{x2:.0f} f(x) dx = {area_1_2:.3f}"),
        go.Scatter(x=np.concatenate([[x2], x_fill_2_3, [x3]]),
                   y=np.concatenate([[0], f(x_fill_2_3), [0]]),
                   fill="toself", fillcolor="green", opacity=0.5,
                   line=dict(color="green"),
                   name=f"∫{x2:.0f}^{x3:.0f} f(x) dx = {area_2_3:.3f}"),
        go.Scatter(x=np.concatenate([[x1], x_fill_1_2, [x2]]),
                   y=np.concatenate([[0], factor*f(x_fill_1_2), [0]]),
                   fill="toself", fillcolor="red", opacity=0.3,
                   line=dict(color="red"),
                   name=f"∫{x1:.0f}^{x2:.0f} {factor}f(x) dx = {area_1_2_factor:.3f}"),
        go.Scatter(x=np.concatenate([[x1], x_fill_1_3, [x3]]),
                   y=np.concatenate([[0], f(x_fill_1_3), [0]]),
                   fill="toself", fillcolor="purple", opacity=0.2,
                   line=dict(color="purple"),
                   name=f"∫{x1:.0f}^{x3:.0f} f(x) dx = {area_1_3:.3f}")
    ]

# --- Crear figura inicial ---
fig = go.Figure(data=make_traces(1, 4, 7, 3))

# --- Valores enteros de los parámetros ---
#x1_vals = np.arange(1, 3+0.5)
#x2_vals = np.arange(4, 6+1)  # empieza en 4
#x3_vals = np.arange(7, 8+1)
#factor_vals = np.arange(1, 3+1)
# --- Valores de los parámetros con paso 0.5 ---
x1_vals = np.arange(1, 3.5, 0.5)   # de 1 a 3, con paso 0.5
x2_vals = np.arange(4, 6.5, 0.5)   # de 4 a 6, con paso 0.5
x3_vals = np.arange(7, 9.5, 0.5)   # de 7 a 8, con paso 0.5
factor_vals = np.arange(1, 3.5, 0.5)  # de 1 a 3, con paso 0.5


# Crear frames para cada parámetro
frames = []
for val in x1_vals:
    frames.append(go.Frame(data=make_traces(val, 4, 7, 3), name=f"x1={val}"))
for val in x2_vals:
    frames.append(go.Frame(data=make_traces(1, val, 7, 3), name=f"x2={val}"))
for val in x3_vals:
    frames.append(go.Frame(data=make_traces(1, 4, val, 3), name=f"x3={val}"))
for val in factor_vals:
    frames.append(go.Frame(data=make_traces(1, 4, 7, val), name=f"factor={val}"))

fig.frames = frames

# --- Sliders ---
sliders = []
for i, (param, vals) in enumerate(zip(["x1","x2","x3","factor"], [x1_vals, x2_vals, x3_vals, factor_vals])):
    steps = [dict(method="animate", args=[[f"{param}={v}"], 
                                          dict(mode="immediate", frame=dict(duration=500, redraw=True), transition=dict(duration=0))],
                  label=f"{v}") for v in vals]
    sliders.append(dict(
        active=0,
        currentvalue={"prefix": f"{param} = "},
        pad={"t":50},
        steps=steps,
        x=0.0 + i*0.25, xanchor='left',
        y=-0.25, yanchor='top',  # bajamos los sliders para que no superpongan
        len=0.22
    ))

# --- Layout ---
fig.update_layout(
    autosize=True,
    height=None, 
    width=None,
    sliders=sliders,
    title="Integrales de áreas entre x1, x2 y x3, y factor",
    xaxis_title="x",
    yaxis_title="y",
    template="plotly_white",
    #height=600,
    updatemenus=[dict(
        type="buttons",
        showactive=False,
        x=-0.1,  # mover Play más a la izquierda
        y=1.05,
        xanchor="left",
        yanchor="top",
        buttons=[
            dict(label="▶ Play",
                 method="animate",
                 args=[None, dict(frame=dict(duration=500, redraw=True),
                                  fromcurrent=True)]),
            dict(label="⏸ Pause",
                 method="animate",
                 args=[[None], dict(frame=dict(duration=0, redraw=False),
                                    mode="immediate")])
        ]
    )]
)
fig.write_html(
    r"_build/html/areas6.html",
    include_plotlyjs="cdn",
    full_html=False
    )
fig.show()
