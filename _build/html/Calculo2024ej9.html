Calculo2024#

Pregunta9#

Enunciado:

Sea la función

\[ f(x) = x^4 - 4x^3 + 5 \]

a) Determinar los valores de \(x\) para los cuales \(f'(x) = 0\).

b) Determinar los intervalos en los cuales \(f(x)\) es creciente y los intervalos en los cuales \(f(x)\) es decreciente.

c) Determinar las coordenadas de los puntos en los cuales \(f(x)\) tiene valores extremos locales (máximos o mínimos relativos).

**Desarrollo **#

Sea la función:

\[ f(x) = x^4 - 4x^3 + 5 \]

1. Derivadas#

\[ f'(x) = 4x^3 - 12x^2 = 4x^2(x-3) \]
\[ f''(x) = 12x(x-2) \]

  • \(f'(x)\) nos ayuda a determinar crecimiento y decrecimiento, y extremos locales.

  • \(f''(x)\) nos ayuda a determinar concavidad y puntos de inflexión.

2. Puntos críticos (donde \(f'(x)=0\))#

\[ 4x^2(x-3) = 0 \implies x = 0, 3 \]

Cuadro de puntos críticos:

Nota: x = 0 no es extremo porque la derivada cambia de signo de negativo a negativo (no cambia de creciente a decreciente).

3. Puntos de inflexión (donde \(f''(x)=0\))#

\[ 12x(x-2) = 0 \implies x=0, 2 \]

4. Intervalos de crecimiento y decrecimiento#

\[ f'(x) = 4x^2(x-3) \]

  • Factor \(4x^2 \ge 0\) siempre (excepto x=0)

6. Extremos locales#

  • Mínimo local: \((3, f(3)) = (3, -22)\)

  • Máximo local: No existe

El punto x = 0 no es extremo local.

8. Observaciones finales#

  1. La función decrece hasta \(x = 3\) y luego crece.

  2. Presenta un mínimo local en \(x=3\).

  3. Cambia de concavidad en \(x=0\) y \(x=2\) (puntos de inflexión).

  4. La concavidad hacia arriba se observa antes de \(x=0\) y después de \(x=2\), mientras que entre \(x=0\) y \(x=2\) la función es cóncava hacia abajo.

Código#

import numpy as np
import plotly.graph_objects as go

# Función y derivadas
def f(x):
    return x**4 - 4*x**3 + 5

def f_prime(x):
    return 4*x**3 - 12*x**2

def f_double_prime(x):
    return 12*x**2 - 24*x  # o 12*x*(x-2)

# Rango de x
x_full = np.linspace(-1, 4, 300)
y_full = f(x_full)
y_prime = f_prime(x_full)
y_double_prime = f_double_prime(x_full)

# Puntos críticos y extremos
x_crit = [0, 3]
y_crit = [f(xi) for xi in x_crit]
labels_crit = ["No extremo", "Mínimo local"]

# Puntos de inflexión
x_inflex = [0, 2]
y_inflex = [f(xi) for xi in x_inflex]

# Concavidad (para colorear puntos)
concave_colors = ["green" if xi<0 or xi>2 else "red" for xi in x_full]

# Crear figura
fig = go.Figure()

# --- Trazas fijas ---
fig.add_trace(go.Scatter(x=x_crit, y=y_crit, mode='markers+text',
                         marker=dict(color='blue', size=10),
                         text=labels_crit, textposition='top right',
                         name='Extremos locales'))

fig.add_trace(go.Scatter(x=x_inflex, y=y_inflex, mode='markers+text',
                         marker=dict(color='orange', size=10),
                         text=["Inflección","Inflección"], textposition='bottom right',
                         name='Puntos de inflexión'))

# --- Trazas animadas vacías ---
fig.add_trace(go.Scatter(x=[], y=[], mode='lines', line=dict(color='black'), name='f(x)'))                   # 0
fig.add_trace(go.Scatter(x=[], y=[], mode='markers', marker=dict(size=4, color=[]), showlegend=False))       # 1
fig.add_trace(go.Scatter(x=[], y=[], mode='lines', line=dict(color='purple', dash='dot'), name="f'(x)"))     # 2
fig.add_trace(go.Scatter(x=[], y=[], mode='lines', line=dict(color='brown', dash='dash'), name="f''(x)"))    # 3

# --- Flechas fijas de crecimiento/decrecimiento ---
annotations = []
growth_intervals = [(-1,3),(3,4)]
for start,end in growth_intervals:
    y_arrow = f(start) + 3
    annotations.append(dict(
        x=(start+end)/2, y=y_arrow,
        ax=start, ay=y_arrow-2,
        xref="x", yref="y",
        axref="x", ayref="y",
        showarrow=True,
        arrowhead=3,
        arrowsize=1,
        arrowwidth=2,
        arrowcolor="blue",
        text="Crecimiento" if start>=3 else "Decrecimiento",
        font=dict(size=10, color="blue")
    ))

fig.update_layout(annotations=annotations)

# --- Frames animados ---
frames = []
frame_step = 5
for i in range(frame_step, len(x_full)+1, frame_step):
    frames.append(go.Frame(
        data=[
            go.Scatter(x=x_full[:i], y=y_full[:i]),                  # f(x)
            go.Scatter(x=x_full[:i], y=y_full[:i], marker=dict(color=concave_colors[:i], size=4)),  # puntos coloreados
            go.Scatter(x=x_full[:i], y=y_prime[:i]),               # f'(x)
            go.Scatter(x=x_full[:i], y=y_double_prime[:i])        # f''(x)
        ],
        traces=[2,3,4,5],
        name=str(i)
    ))

fig.frames = frames

# --- Slider ---
slider_steps = []
for i, frame in enumerate(frames):
    slider_steps.append(dict(
        method="animate",
        args=[[frame.name],
              {"mode":"immediate",
               "frame":{"duration":50, "redraw":True},
               "transition":{"duration":0}}],
        label=str(i*frame_step)
    ))

sliders = [dict(
    active=0,
    currentvalue={"prefix":"x hasta: "},
    pad={"t":50},
    steps=slider_steps
)]

# --- Layout con botones ---
fig.update_layout(
    title="f(x), f'(x) y f''(x) animadas con puntos y flechas",
    xaxis_title="x",
    yaxis_title="y",
    template="plotly_white",
    sliders=sliders,
    updatemenus=[dict(
        type="buttons",
        showactive=False,
        x=-0.1,
        y=1.0,
        xanchor="left",
        yanchor="top",
        buttons=[
            dict(label="Play",
                 method="animate",
                 args=[None, {"frame": {"duration":50, "redraw":True},
                              "fromcurrent": True, "transition": {"duration":0}}]),
            dict(label="Pause",
                 method="animate",
                 args=[[None], {"frame": {"duration":0, "redraw":False},
                                "mode":"immediate"}])
        ]
    )]
)

fig.show()

import numpy as np
import plotly.graph_objects as go

# Función y derivadas
def f(x):
    return x**4 - 4*x**3 + 5

def f_prime(x):
    return 4*x**3 - 12*x**2

def f_double_prime(x):
    return 12*x**2 - 24*x

# Rango de x
x_full = np.linspace(-1, 4.1, 300)
y_full = f(x_full)
y_prime = f_prime(x_full)
y_double = f_double_prime(x_full)

# Crear figura
fig = go.Figure()

# --- Traza animada de f(x) ---
fig.add_trace(go.Scatter(x=[], y=[], mode='lines', line=dict(color='black'), name='f(x)'))

# --- Traza animada de f'(x) ---
fig.add_trace(go.Scatter(x=[], y=[], mode='lines', line=dict(color='purple', dash='dot'), name="f'(x)"))

# --- Traza animada de f''(x) ---
fig.add_trace(go.Scatter(x=[], y=[], mode='lines', line=dict(color='brown', dash='dash'), name="f''(x)"))

# --- Puntos críticos fijos ---
x_crit = [0, 3]
y_crit = [f(xi) for xi in x_crit]
labels_crit = ["No extremo", "Mínimo local"]
fig.add_trace(go.Scatter(
    x=x_crit, y=y_crit, mode='markers+text',
    marker=dict(color='blue', size=10),
    text=labels_crit, textposition='top right',
    name='Extremos locales'
))

# --- Punto(s) de inflexión fijos ---
x_inflex = [0, 2]  # puedes agregar más si hay
y_inflex = [f(xi) for xi in x_inflex]
labels_inflex = ["Inflexión"]*len(x_inflex)
fig.add_trace(go.Scatter(
    x=x_inflex, y=y_inflex, mode='markers+text',
    marker=dict(color='orange', size=10),
    text=labels_inflex, textposition='bottom right',
    name='Puntos de inflexión'
))

# --- Frames ---
frames = []
frame_step = 5
for i in range(frame_step, len(x_full)+1, frame_step):
    x_cur = x_full[:i]
    y_cur = y_full[:i]
    y_prime_cur = y_prime[:i]
    y_double_cur = y_double[:i]
    # Dentro del bucle de frames
    #y_prime_cur_clip = np.clip(y_prime_cur, -20, 20)      # limita entre -20 y 20
    #y_prime_cur_limited = y_prime_cur[y_prime_cur <= 10]  # solo valores <= 20
    #x_prime_cur_limited = x_cur[:len(y_prime_cur_limited)]

    #y_double_cur_clip = np.clip(y_double_cur, -20, 20)    # limita entre -20 y 20
    #y_double_cur_limited = y_double_cur[y_double_cur <= 10]
    #x_double_cur_limited = x_cur[:len(y_double_cur_limited)]
    
    mask_double = y_double_cur <= 10
    x_double_cur_limited = x_cur[mask_double]
    y_double_cur_limited = y_double_cur[mask_double]

    mask_prime = y_prime_cur <= 10
    x_prime_cur_limited = x_cur[mask_prime]
    y_prime_cur_limited = y_prime_cur[mask_prime]

    # Indicadores numéricos
    deriv_val = y_prime_cur[-1]
    double_deriv_val = y_double_cur[-1]
    annotation = [dict(
        x=0.62, y=0.95,
        xref='paper', yref='paper',
        text=f"f'({x_cur[-1]:.2f}) = {deriv_val:.2f}<br>f''({x_cur[-1]:.2f}) = {double_deriv_val:.2f}",
        showarrow=False,
        font=dict(size=14, color="black"),
        align='left',
        bordercolor="black",
        borderwidth=1,
        borderpad=4,
        bgcolor="white",
        opacity=0.9
    )]
    
    frames.append(go.Frame(
        data=[
            go.Scatter(x=x_cur, y=y_cur),          # f(x)
            #go.Scatter(x=x_cur, y=y_prime_cur),    # f'(x)
           # go.Scatter(x=x_cur, y=y_double_cur)    # f''(x)
            #go.Scatter(x=x_cur, y=y_prime_cur_clip),       # f'(x) limitada a 20
            #go.Scatter(x=x_prime_cur_limited, y=y_prime_cur_limited),  # f'(x) hasta 20
           # go.Scatter(x=x_cur, y=y_double_cur_clip)       # f''(x) limitada a 20
            #go.Scatter(x=x_double_cur_limited, y=y_double_cur_limited) # f''(x) hasta 20
            go.Scatter(x=x_prime_cur_limited, y=y_prime_cur_limited),
            go.Scatter(x=x_double_cur_limited, y=y_double_cur_limited)
        ],
        layout=dict(annotations=annotation),
        name=str(i)
    ))

fig.frames = frames

# --- Slider ---
slider_steps = []
for i, frame in enumerate(frames):
    slider_steps.append(dict(
        method="animate",
        args=[[frame.name], {"mode":"immediate","frame":{"duration":50,"redraw":True},"transition":{"duration":0}}],
        label=str(i*frame_step)
    ))

sliders = [dict(active=0, currentvalue={"prefix":"x hasta: "}, pad={"t":50}, steps=slider_steps)]

# --- Layout con botones Play/Pause ---
fig.update_layout(
    autosize=True,
    width=None,
    height=None,
    title="f(x), f'(x) y f''(x) animadas con indicadores numéricos y puntos fijos",
    xaxis_title="x",
    yaxis_title="y",
    yaxis=dict(range=[min(y_full)-1, 10]),  # Todos los valores por encima de 20 no se mostrarán
    template="plotly_white",
    sliders=sliders,
    updatemenus=[dict(
        type="buttons",
        showactive=False,
        x=-0.1, y=1.0,
        xanchor="left", yanchor="top",
        buttons=[
            dict(label="Play",
                 method="animate",
                 args=[None, {"frame": {"duration":50, "redraw":True}, "fromcurrent": True, "transition": {"duration":0}}]),
            dict(label="Pause",
                 method="animate",
                 args=[[None], {"frame": {"duration":0, "redraw":False}, "mode":"immediate"}])
        ]
    )]
)

fig.write_html(
    r"_build/html/derivadacurva2.html",
    include_plotlyjs="cdn",
    auto_open=False,
    full_html=False
)

fig.show()
